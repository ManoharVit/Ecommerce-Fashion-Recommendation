def create_age_interval(x):
   
    if x <= 20:
        return [16, 20]
    elif x <= 30:
        return [21, 30]
    elif x <= 40:
        return [31, 40]
    elif x <= 50:
        return [41, 50]
    elif x <= 60:
        return [51, 60]
    else:
        return [61, 99]
